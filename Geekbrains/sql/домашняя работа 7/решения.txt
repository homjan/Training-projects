1. Создайте двух пользователей которые имеют доступ к базе данных shop. Первому пользователю shop_read должны быть доступны только запросы на чтение данных, второму пользователю shop — любые операции в пределах базы данных shop.

use shop;

DROP USER shop_read;

CREATE USER shop_read IDENTIFIED WITH Sha256_password BY 'Pass_1234';


GRANT SELECT ON * TO shop_read;
// Выбираем все таблицы базы данных shop 
select user() shop_read;

Перезапускаем сессию, выбираем ранее созданные пользователь-пароль

select * from catalogs;

insert into catalogs values(6, 'Клавиатуры');

Убеждаемся что прав не достаточно. (ошибка на команде insert)


use shop;
